**Example normalizer skeleton** (mirror `payloadString` style):

```typescript
function pickString(obj: Record<string, unknown>, keys: string[]): string | undefined {
    for (const key of keys) {
        const v = obj[key];
        if (typeof v === 'string' && v.length > 0) return v;
    }
    return undefined;
}

export function normalizeTaskArgs(args: Record<string, unknown>): Record<string, unknown> {
    const description = pickString(args, ['description', 'task_description']);
    const prompt = pickString(args, ['prompt', 'message']);
    return {
        ...args,
        ...(description ? { description } : {}),
        ...(prompt ? { prompt } : {}),
    };
}
```

### `hub/src/sync/toolCallProjection.ts` — `inferToolNameFromPayload` extension (service, transform)

**Guard + early returns pattern** (lines 105-134):

```typescript
export function inferToolNameFromPayload(input: unknown, output: unknown): string | null {
    const inObj = isObject(input) ? input : null;
    const outObj = isObject(output) ? output : null;

    if (inObj) {
        if (payloadString(inObj, ['command', 'cmd'])) return 'Bash';
        // ... existing Grep/Glob/Read/Write/Edit/TodoWrite ...

        // 01.3: advanced tools — use CANONICAL wire fields (post-CLI-normalize)
        if (Array.isArray(inObj.questions) && inObj.questions.length > 0) return 'AskUserQuestion';
        if (payloadString(inObj, ['skill'])) return 'Skill';
        if (payloadString(inObj, ['notebook_path'])) {
            if (payloadString(inObj, ['edit_mode']) || 'old_string' in inObj) return 'NotebookEdit';
            return 'NotebookRead';
        }
        const description = payloadString(inObj, ['description']);
        const teamName = payloadString(inObj, ['team_name']);
        if (description || teamName) return 'Task';
        if (payloadString(inObj, ['prompt']) && payloadString(inObj, ['subagent_type'])) return 'Agent';
        if (payloadString(inObj, ['prompt'])) return 'Agent'; // ambiguous → Agent per RESEARCH
    }
    return null;
}
```

**Placeholder upgrade** (lines 136-143) — no change to call site:

    if (!isPlaceholderProjectionName(candidate)) return candidate;
