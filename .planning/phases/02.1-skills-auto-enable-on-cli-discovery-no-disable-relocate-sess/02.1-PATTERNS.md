**Single-flight refresh pattern** (RESEARCH Pattern 1):

```typescript
let cachedSkills: SkillSummary[] = [];
let refreshInFlight: Promise<SkillSummary[]> | null = null;

function refreshSkills(): Promise<SkillSummary[]> {
    if (!refreshInFlight) {
        refreshInFlight = listSkills(workingDirectory)
            .then((skills) => {
                cachedSkills = skills;
                return skills;
            })
            .catch((err) => {
                logger.debug('[cursor] listSkills failed; using last cache', err);
                return cachedSkills;
            })
            .finally(() => {
                refreshInFlight = null;
            });
    }
    return refreshInFlight;
}

session.onUserMessage(async (message, localId) => {

    const formattedText = formatMessageWithAttachments(message.content.text, message.content.attachments);

});

const { getSuggestions: getSkillSuggestions } = useSkills(api, sessionId)

const getAutocompleteSuggestions = useCallback(async (query: string) => {
    if (!query.startsWith('/')) {
        return await getSlashSuggestions(query)
    }
    const [skills, commands] = await Promise.all([
        getSkillSuggestions(query),
        getSlashSuggestions(query),

    const seen = new Set<string>()
    return [...skills, ...commands].filter((s) => {

        seen.add(s.key)
        return true
    })
}, [getSkillSuggestions, getSlashSuggestions])
```
